108void BLESetup() {
109
110 if (!BLE.begin()) {
111 Serial.println("BLE initialization failed!");
112 while (1);
113 }
114
115 BLE.setDeviceName("M5Core2_BLE_Device");
116 BLE.setLocalName("M5Core2");
117 BLE.setAdvertisedService(M5CoreService);
118
119 M5CoreService.addCharacteristic(M5CoreCharacteristic);
120
121 BLE.addService(M5CoreService);
122 BLE.advertise();
123
124 SERIAL_PRINTLN("En attente de connexion BLE...");
125
126 queueBLERead = xQueueCreate(10, TRAME_SIZE);
127 xTaskCreatePinnedToCore(taskBLERead, "BLE_Read", 8192, nullptr, 2, nullptr, 1);
128
129 queueBLEWrite = xQueueCreate(10, TRAME_SIZE);
130 xTaskCreatePinnedToCore(taskBLEWrite, "BLE_Write", 8192, nullptr, 2, nullptr, 1);
131}

163bool sendDataBLE(BLEDevice& central, const char* serviceUuid, const char* characteristicUuid, const char* dataToSend) {
164 // Découvrir le service BLE spécifié
165 if (central.discoverService(serviceUuid)) {
166 SERIAL_PRINTLN("Service trouvé !");
167
168 // Trouver la caractéristique associée
169 BLECharacteristic characteristic = central.characteristic(characteristicUuid);
170 if (characteristic) {
171 // Tenter d'envoyer les données
172 if (characteristic.writeValue((uint8_t*)dataToSend, strlen(dataToSend))) {
173 SERIAL_PRINTLN("Données envoyées avec succès !");
174 return true; // Succès
175 } else {
176 SERIAL_PRINTLN("Échec de l'envoi des données.");
177 }
178 } else {
179 SERIAL_PRINTLN("Caractéristique non trouvée !");
180 }
181 } else {
182 SERIAL_PRINTLN("Service non trouvé !");
183 }
184 return false; // Échec
185}

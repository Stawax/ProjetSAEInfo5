101void BLESetup() {
102
103 if (!BLE.begin()) {
104 Serial.println("BLE initialization failed!");
105 while (1);
106 }
107
108 /* Nom de votre périphérique */
109 BLE.setDeviceName("MStamp_BLE_Device");
110 /* Nom local BLE */
111 BLE.setLocalName("M5Stack");
112 BLE.setAdvertisedService(M5StackService);
113
114 M5StackService.addCharacteristic(M5StackCharacteristic);
115
116 BLE.addService(M5StackService);
117
118 queueBLERead = xQueueCreate(10, TRAME_SIZE);
119 xTaskCreatePinnedToCore(taskBLERead, "BLE_Read", 8192, nullptr, 2, nullptr, 1);
120
121 queueBLEWrite = xQueueCreate(10, TRAME_SIZE);
122 xTaskCreatePinnedToCore(taskBLEWrite, "BLE_Write", 8192, nullptr, 2, nullptr, 1);
123}

154bool sendDataBLE(BLEDevice& central, const char* serviceUuid, const char* characteristicUuid, const char* dataToSend) {
155 // Découvrir le service BLE spécifié
156 if (central.discoverService(serviceUuid)) {
157 SERIAL_PRINTLN("Service trouvé !");
158
159 // Trouver la caractéristique associée
160 BLECharacteristic characteristic = central.characteristic(characteristicUuid);
161 if (characteristic) {
162 // Tenter d'envoyer les données
163 if (characteristic.writeValue((uint8_t*)dataToSend, strlen(dataToSend))) {
164 SERIAL_PRINTLN("Données envoyées avec succès !");
165 return true; // Succès
166 } else {
167 SERIAL_PRINTLN("Échec de l'envoi des données.");
168 }
169 } else {
170 SERIAL_PRINTLN("Caractéristique non trouvée !");
171 }
172 } else {
173 SERIAL_PRINTLN("Service non trouvé !");
174 }
175 return false; // Échec
176}

85void loop() {
86
87#if BLE_MODE
88 SERIAL_PRINTLN("Recherche de périphérique M5Core2...");
89
90 central = scanForDevice("M5Core2");
91
92 // Si le périphérique est trouvé
93 if (central.localName().length() > 0) {
94 led.show(255, 255, 255);
95
96 while (true) {
97 if (digitalRead(BOUTON) == 0) {
98
99 // Tenter la connexion avec le périphérique trouvé
100 if (central.connect()) {
101
102 led.show(255, 0, 255);
103 SERIAL_PRINTLN("Connecté à M5Core2 !");
104
105 // Préparer les données à envoyer
106 String uid = generateRandomUID();
107 String roomName = generateRandomRoomName();
108 int deviceType = generateRandomDeviceType();
109
110 static char dataToSend[100];
111 snprintf(dataToSend,
112 sizeof(dataToSend),
113 "{\"action\":2,\"idDevice\":1,\"idTypeDevice\":%d,\"uid\":\"%s\",\"nom\":\"%s\"}",
114 deviceType, uid.c_str(), roomName.c_str());
115
116 // Ajouter le message dans la file BLEWrite
117 xQueueSendToBack(queueBLEWrite, (void *)dataToSend, portMAX_DELAY);
118
119 // Attendre que la connexion soit terminée ou que 60 seconde passe ou la réception d'une info
120 unsigned long startTime = millis();
121 while (central.connected() && millis() - startTime < 60000 && information == ActionInformation::NO_INFO) {
122 BLE_callback();
123 delay(100);
124 }
125
126 if(information == ActionInformation::VALID) {
127 SERIAL_PRINTLN("M5Stack validé par le M5Core !");
128
129 information = ActionInformation::NO_INFO;
130 led.show(0, 255, 0);
131 delay(5000);
132 } else {
133 SERIAL_PRINTLN("M5Stack refusé par le M5Core !");
134
135 information = ActionInformation::NO_INFO;
136 led.show(255, 0, 0);
137 delay(5000);
138 }
139
140 // Déconnexion une fois la communication terminée
141 central.disconnect();
142
143 SERIAL_PRINTLN("Déconnecté de M5Core2.");
144 } else {
145 SERIAL_PRINTLN("Échec de la connexion à M5Core2.");
146 }
147
148 // Réinitialiser la LED à bleue après l'opération
149 led.show(0, 0, 255);
150 break; // Quitter la boucle dès qu'une opération est effectuée
151 }
152 }
153 }
154 delay(1000);
155#endif
156}

164BLEDevice scanForDevice(const char* targetName) {
165 BLEDevice foundDevice;
166
167 while (true) { // Boucle infinie jusqu'à ce que le périphérique soit trouvé
168 BLE.scan();
169
170 while (BLE.available()) {
171 foundDevice = BLE.available();
172
173 SERIAL_PRINTLN("======================");
174 SERIAL_PRINT("Nom du périphérique : ");
175 SERIAL_PRINTLN(foundDevice.localName());
176 SERIAL_PRINT("Adresse : ");
177 SERIAL_PRINTLN(foundDevice.address());
178 SERIAL_PRINT("RSSI : ");
179 SERIAL_PRINTLN(foundDevice.rssi());
180
181 // Vérifier si le nom correspond à celui recherché
182 if (strcmp(foundDevice.localName().c_str(), targetName) == 0) {
183 SERIAL_PRINTLN(targetName);
184 SERIAL_PRINTLN(" trouvé...");
185 BLE.stopScan(); // Arrêter le scan après avoir trouvé le périphérique
186 return foundDevice;
187 }
188 }
189
190 delay(1000); // Pause de 1 seconde avant de relancer un nouveau scan
191 }
192}

200String generateRandomUID() {
201 char uid[18];
202
203 for (int i = 0; i < 6; ++i) {
204 char highNibble = "0123456789ABCDEF"[random(0, 16)];
205 char lowNibble = "0123456789ABCDEF"[random(0, 16)];
206
207 uid[i * 3] = highNibble; // Premier caractère
208 uid[i * 3 + 1] = lowNibble; // Deuxième caractère
209
210 if (i < 5) {
211 uid[i * 3 + 2] = ':';
212 } else {
213 uid[i * 3 + 2] = '\0'; // Pour terminer la chaîne après le dernier segment
214 }
215 }
216
217 SERIAL_PRINTLN(uid);
218 return String(uid);
219}

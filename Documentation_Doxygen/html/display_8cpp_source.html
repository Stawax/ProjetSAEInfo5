38void taskUpdateDisplay(void *pvParameters) {
39 // Initialisation de l'état actuel de l'écran et de l'action reçue
40 StateScreen actualStateScreen = HOME_SCREEN;
41 ActionPress receivedAction = ActionPress::NO_PRESS;
42
43 // Initialisation de l'objet Device pour stocker l'état actuel des nouveaux appareils
44 Device actualNewDevices;
45 actualNewDevices.reset();
46
47 // Délai entre chaque mise à jour (100 ms)
48 const TickType_t xDelay = pdMS_TO_TICKS(100);
49 TickType_t lastTimeUpdate = xTaskGetTickCount();
50
51 // Affiche l'écran d'accueil au démarrage
52 displayScreenHome();
53
54 // Boucle principale de la tâche
55 while(1) {
56 // Vérifie si l'état de l'écran a changé, si oui affiche le nouvelle écran.
57 if (actualStateScreen != screen) {
58 actualStateScreen = screen;
59 switch (actualStateScreen) {
60 case HOME_SCREEN:
61 displayScreenHome();
62 break;
63 case ADD_SCREEN:
64 displayScreenAdd();
65 if (!new_devices.empty()) {
66 displayDevice(new_devices[currentDeviceIndex], TEXT_COLOR, BACK_COLOR);
67 displayIcon(cross_button_bits, 30, HEIGHT - 50, ICONS_WIDTH, ICONS_HEIGHT, TEXT_COLOR, BACK_COLOR);
68 displayIcon(valid_button_bits, WIDTH - 70, HEIGHT - 50, ICONS_WIDTH, ICONS_HEIGHT, TEXT_COLOR, BACK_COLOR);
69 }
70 break;
71 case MENU_SCREEN:
72 displayScreenMenu();
73 if (!devices.empty()) {
74 displayDevice(devices[currentDeviceIndex], TEXT_COLOR, BACK_COLOR);
75 }
76 break;
77 default:
78 break;
79 }
80 }
81
82 // Gestion des actions utilisateur dans l'écran "ADD_SCREEN"
83 if (actualStateScreen == ADD_SCREEN) {
84 if (xQueueReceive(actionQueue, &receivedAction, 0)) {
85 switch (receivedAction) {
86 case ActionPress::LEFT_PRESS:
87 if(!new_devices.empty()) {
88 currentNewDeviceIndex = (currentNewDeviceIndex - 1 + new_devices.size()) % new_devices.size();
89 } else {
90 currentNewDeviceIndex = 0;
91 }
92 break;
93 case ActionPress::RIGHT_PRESS:
94 if(!new_devices.empty()) {
95 currentNewDeviceIndex = (currentNewDeviceIndex + 1) % new_devices.size();
96 } else {
97 currentNewDeviceIndex = 0;
98 }
99 break;
100 case ActionPress::VALIDATE_PRESS:
101 validateDevice(currentNewDeviceIndex);
102 if(!new_devices.empty()) {
103 currentNewDeviceIndex = currentNewDeviceIndex % new_devices.size();
104 } else {
105 currentNewDeviceIndex = 0;
106 }
107 break;
108 case ActionPress::DENIED_PRESS:
109 deniedDevice(currentNewDeviceIndex);
110 if(!new_devices.empty()) {
111 currentNewDeviceIndex = currentNewDeviceIndex % new_devices.size();
112 } else {
113 currentNewDeviceIndex = 0;
114 }
115 break;
116 default:
117 break;
118 }
119 }
120
121 // Mise à jour de l'écran si le new device a changé
122 if (!new_devices.empty() && actualNewDevices.uid != new_devices[currentNewDeviceIndex].uid) {
123 actualNewDevices = new_devices[currentNewDeviceIndex];
124 displayDevice(actualNewDevices, TEXT_COLOR, BACK_COLOR);
125 displayTime(actualNewDevices, TEXT_COLOR, BACK_COLOR);
126 displayIcon(cross_button_bits, 30, HEIGHT - 50, ICONS_WIDTH, ICONS_HEIGHT, TEXT_COLOR, BACK_COLOR);
127 displayIcon(valid_button_bits, WIDTH - 70, HEIGHT - 50, ICONS_WIDTH, ICONS_HEIGHT, TEXT_COLOR, BACK_COLOR);
128 } else if (new_devices.empty() && !actualNewDevices.isReset()) {
129 actualNewDevices.reset();
130 refreshPartScreen(CENTER, TEXT_COLOR, TFT_BLACK);
131 }
132 }
133
134 // Gestion des actions utilisateur dans l'écran "MENU_SCREEN"
135 if (actualStateScreen == MENU_SCREEN) {
136 if (xQueueReceive(actionQueue, &receivedAction, 0)) {
137 switch (receivedAction) {
138 case ActionPress::LEFT_PRESS:
139 currentDeviceIndex = (currentDeviceIndex - 1 + devices.size()) % devices.size();
140 if (!devices.empty()) {
141 displayDevice(devices[currentDeviceIndex], TEXT_COLOR, BACK_COLOR);
142 }
143 break;
144 case ActionPress::RIGHT_PRESS:
145 currentDeviceIndex = (currentDeviceIndex + 1) % devices.size();
146 if (!devices.empty()) {
147 displayDevice(devices[currentDeviceIndex], TEXT_COLOR, BACK_COLOR);
148 }
149 break;
150 default:
151 break;
152 }
153 }
154 }
155
156 // Mise à jour de automatique de l'écran toutes les secondes
157 if (xTaskGetTickCount() - lastTimeUpdate >= pdMS_TO_TICKS(1000)) {
158 lastTimeUpdate = xTaskGetTickCount(); // Met à jour le dernier temps
159 switch (actualStateScreen) {
160 case ADD_SCREEN:
161 if (!new_devices.empty()) {
162 displayTime(new_devices[currentNewDeviceIndex], TEXT_COLOR, BACK_COLOR);
163 }
164 break;
165 default:
166 break;
167 }
168 }
169
170 // Délai avant la prochaine mise à jour
171 vTaskDelay(xDelay);
172 }
173}

282void displayDevice(const Device& device, uint16_t textColor, uint16_t backColor) {
283 refreshPartScreen(CENTER, TEXT_COLOR, TFT_BLACK);
284
285 int textX = 20;
286 int textY = HEIGHT - 160;
287 int lineSpacing = 30;
288
289 M5.Lcd.setTextColor(textColor);
290 M5.Lcd.setTextSize(2);
291
292 M5.Lcd.setCursor(textX, textY);
293 M5.Lcd.printf("UID: %s", device.uid.c_str());
294 M5.Lcd.setCursor(textX, textY + lineSpacing);
295 M5.Lcd.printf("Nom: %s", device.name.c_str());
296
297 // Conversion du type de capteur en texte
298 std::string typeString = convertSensorType(device.idTypeDevice - 1);
299 M5.Lcd.setCursor(textX, textY + 2*lineSpacing);
300 M5.Lcd.printf("Type: %s", typeString.c_str());
301}

311void displayTime(const Device& device, uint16_t textColor, uint16_t backColor) {
312 M5.Lcd.fillRect(140, HEIGHT - 30, 50, 30, backColor);
313
314 int textX = 140;
315 int textY = HEIGHT - 30;
316
317 uint32_t elapsedMillis = millis() - device.timestamp;
318 uint32_t elapsedSeconds = (elapsedMillis / 1000) % 60; // Limiter à 60 secondes
319
320 M5.Lcd.setTextColor(textColor);
321 M5.Lcd.setTextSize(2);
322 M5.Lcd.setCursor(textX, textY);
323 M5.Lcd.printf("%u s", elapsedSeconds);
324}

338void displayIcon(const uint8_t *bitmap, int x, int y, int w, int h, uint16_t frontColor, uint16_t backColor) {
339 uint16_t color;
340 for (int j = 0; j < h; j++) {
341 for (int i = 0; i < w; i++) {
342 color = bitmap[j * w + i] == 1 ? frontColor : backColor;
343 M5.Lcd.drawPixel(x + i, y + j, color);
344 }
345 }
346}
